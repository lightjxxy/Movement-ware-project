function Utility:Create(class, properties)
    local instance = Instance.new(class)
    for k, v in pairs(properties) do
        instance[k] = v
    end
    return instance
end

function Utility:MakeDraggable(topbar, window)
    local dragging
    local dragInput
    local dragStart
    local startPos

    topbar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = window.Position
            
            local moveConn, endConn
            
            moveConn = UserInputService.InputChanged:Connect(function(moveInput)
                if moveInput.UserInputType == Enum.UserInputType.MouseMovement or moveInput.UserInputType == Enum.UserInputType.Touch then
                    local delta = moveInput.Position - dragStart
                    window.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
                end
            end)
            
            endConn = input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                    if moveConn then moveConn:Disconnect() end
                    if endConn then endConn:Disconnect() end
                end
            end)
        end
    end)
end

    local function OpenModal(title, setupFunc, sourceElement)
        PopupTitle.Text = title
        
        for _, conn in ipairs(PopupConnections) do
            if conn.Disconnect then conn:Disconnect() end
        end
        table.clear(PopupConnections)

        for _, child in pairs(PopupContent:GetChildren()) do
            if not child:IsA("UIListLayout") then child:Destroy() end
        end
        setupFunc(PopupContent)
        
        -- Positioning logic
        if sourceElement then
            local absPos = sourceElement.AbsolutePosition
            local absSize = sourceElement.AbsoluteSize
            local screenX = ScreenGui.AbsoluteSize.X
            local screenY = ScreenGui.AbsoluteSize.Y
            
            local targetX = absPos.X + absSize.X + 10
            local targetY = absPos.Y
            
            if targetX + 220 > screenX then
                targetX = absPos.X - 220 - 10
            end
            if targetY + PopupFrame.AbsoluteSize.Y > screenY then
                targetY = screenY - PopupFrame.AbsoluteSize.Y - 10
            end
            
            PopupFrame.Position = UDim2.new(0, targetX, 0, targetY)
        else
            PopupFrame.Position = UDim2.new(0.5, -110, 0.5, -140)
        end
        PopupBg.Visible = true
    end

            end

                        end)

                UserInputService.InputBegan:Connect(function(input, gp)
                    if isBinding then

                            currentBind = nil
                            if activeBindBtnInModal then activeBindBtnInModal.Text = "[Click to Bind]" end
                        elseif input.KeyCode ~= Enum.KeyCode.Unknown then
                            currentBind = input.KeyCode
                            if activeBindBtnInModal then activeBindBtnInModal.Text = "[" .. input.KeyCode.Name .. "]" end
                        end
                        isBinding = false
                    elseif not gp and currentBind and input.KeyCode == currentBind then
                        if bindMode == "Toggle" then
                            state = not state
                            TweenService:Create(ToggleBox, TweenInfo.new(0.2), {BackgroundColor3 = state and Theme.ToggleActive or Theme.ToggleInactive}):Play()
                            callback(state)
                        elseif bindMode == "Hold" then
                            state = true
                            TweenService:Create(ToggleBox, TweenInfo.new(0.2), {BackgroundColor3 = Theme.ToggleActive}):Play()
                            callback(state)
                        end
                    end
                end)

                UserInputService.InputEnded:Connect(function(input, gp)
                    if not gp and currentBind and input.KeyCode == currentBind and bindMode == "Hold" then
                        state = false
                        TweenService:Create(ToggleBox, TweenInfo.new(0.2), {BackgroundColor3 = Theme.ToggleInactive}):Play()
                        callback(state)
                    end
                end)

                ToggleBox.MouseButton1Click:Connect(function()
                    state = not state
                    TweenService:Create(ToggleBox, TweenInfo.new(0.2), {BackgroundColor3 = state and Theme.ToggleActive or Theme.ToggleInactive}):Play()
                    callback(state)
                end)
                task.spawn(function() callback(state) end)
            end

                    local pos = math.clamp((input.Position.X - SliderBack.AbsolutePosition.X) / SliderBack.AbsoluteSize.X, 0, 1)
                    SliderFill.Size = UDim2.new(pos, 0, 1, 0)
                    local value = math.floor(min + ((max - min) * pos))
                    ValueLabel.Text = tostring(value)
                    callback(value)
                end

                SliderBtn.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        dragging = true
                        update(input)
                        
                        local moveConn, endConn
                        moveConn = UserInputService.InputChanged:Connect(function(moveInput)
                            if moveInput.UserInputType == Enum.UserInputType.MouseMovement then
                                update(moveInput)
                            end
                        end)

                            if input.UserInputState == Enum.UserInputState.End then
                                dragging = false
                                if moveConn then moveConn:Disconnect() end
                                if endConn then endConn:Disconnect() end
                            end
                        end)
                    end
                end)
                
                task.spawn(function() callback(default) end)
            end

                opts = opts or {}
                local dropText = opts.Name or "Multi Dropdown"
                local list = opts.Options or {}
                local callback = opts.Callback or function() end
                
                local isOpen = false
                local selected = {}
                if opts.Default then
                    for _, v in ipairs(opts.Default) do
                        table.insert(selected, v)
                    end
                end

                local DropdownFrame = Utility:Create("Frame", {
                    Parent = ContentContainer, BackgroundTransparency = 1, Size = UDim2.new(1, -24, 0, 40)
                })

                Utility:Create("TextLabel", {

                })
                Utility:Create("UICorner", { Parent = MainBox, CornerRadius = UDim.new(0, 6) })

                local function getSelectedText()
                    if #selected == 0 then return "None" end
                    return table.concat(selected, ", ")
                end

                local SelectedLabel = Utility:Create("TextLabel", {
                    Parent = MainBox, BackgroundTransparency = 1, Position = UDim2.new(0, 8, 0, 0), Size = UDim2.new(1, -25, 1, 0), Font = Theme.Font, Text = getSelectedText(), TextColor3 = Theme.TextMain, TextSize = 13, TextXAlignment = Enum.TextXAlignment.Left, TextTruncate = Enum.TextTruncate.AtEnd
                })

                local Arrow = Utility:Create("TextLabel", {
                    Parent = MainBox, BackgroundTransparency = 1, Position = UDim2.new(1, -20, 0, 0), Size = UDim2.new(0, 20, 1, 0), Font = Theme.Font, Text = "v", TextColor3 = Theme.TextMain, TextSize = 13
                })

                local DropContainer = Utility:Create("Frame", {
                    Parent = DropdownFrame, BackgroundColor3 = Theme.ElementBg, BorderSizePixel = 0, Position = UDim2.new(0, 0, 0, 44), Size = UDim2.new(1, 0, 0, 0), Visible = false, ClipsDescendants = true
                })
                Utility:Create("UICorner", { Parent = DropContainer, CornerRadius = UDim.new(0, 6) })

                local DropLayout = Utility:Create("UIListLayout", { Parent = DropContainer, SortOrder = Enum.SortOrder.LayoutOrder })

                local function createList()
                    for _, v in pairs(DropContainer:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
                    local h = 0
                    for _, option in ipairs(list) do
                        local OptBtn = Utility:Create("TextButton", {
                            Parent = DropContainer, BackgroundTransparency = 1, Size = UDim2.new(1, 0, 0, 25), Font = Enum.Font.Gotham, Text = "  " .. option, TextColor3 = Theme.TextMain, TextSize = 14, TextXAlignment = Enum.TextXAlignment.Left
                        })
                        h = h + 25

                        OptBtn.MouseButton1Click:Connect(function()
                            local isSel = false
                            for _, v in ipairs(selected) do if v == option then isSel = true break end end

                            if isSel then
                                for i, v in ipairs(selected) do
                                    if v == option then table.remove(selected, i) break end
                                end
                            else
                                table.insert(selected, option)
                            end
                            SelectedLabel.Text = getSelectedText()
                            callback(selected)
                            
                            -- Update all colors
                            for _, btn in pairs(DropContainer:GetChildren()) do
                                if btn:IsA("TextButton") then
                                    local optName = btn.Text:sub(3) -- remove "  " prefix
                                    local sel = false
                                    for _, v in ipairs(selected) do if v == optName then sel = true break end end
                                    btn.TextColor3 = sel and Theme.ToggleActive or Theme.TextMain
                                    btn.Font = sel and Theme.Font or Enum.Font.Gotham
                                end
                            end
                        end)
                    end
                    DropContainer.Size = UDim2.new(1, 0, 0, h)
                    
                    -- Initial color setup
                    for _, btn in pairs(DropContainer:GetChildren()) do
                        if btn:IsA("TextButton") then
                            local optName = btn.Text:sub(3)
                            local sel = false
                            for _, v in ipairs(selected) do if v == optName then sel = true break end end
                            btn.TextColor3 = sel and Theme.ToggleActive or Theme.TextMain
                            btn.Font = sel and Theme.Font or Enum.Font.Gotham
                        end
                    end
                end

                createList()

                table.insert(AccentUpdates, function(col)
                    for _, btn in pairs(DropContainer:GetChildren()) do
                        if btn:IsA("TextButton") then
                            local optName = btn.Text:sub(3)
                            local sel = false
                            for _, v in ipairs(selected) do if v == optName then sel = true break end end
                            if sel then
                                btn.TextColor3 = col
                            end
                        end
                    end
                end)

                MainBox.MouseButton1Click:Connect(function()
                    isOpen = not isOpen
                    DropContainer.Visible = isOpen
                    Arrow.Text = isOpen and "^" or "v"
                    if isOpen then
                        DropdownFrame.Size = UDim2.new(1, -24, 0, 55 + DropContainer.Size.Y.Offset)
                    else
                        DropdownFrame.Size = UDim2.new(1, -24, 0, 50)
                    end
                end)
                
                task.spawn(function() callback(selected) end)
            end

            return Elements
        end

        return TabObj
    end

-- Edge Trimp logic
local edgeTrimpEnabled = false
local LAUNCH_POWER = 100
local hasTriggered = false
local lastVelocity = Vector3.zero

local function getFeetPosition(hrp)
    if not hrp then return Vector3.zero end
    return hrp.Position - Vector3.new(0, 3.5, 0)
end

local function checkEdgeCollision(char, hrp, hum)
    if hasTriggered or not edgeTrimpEnabled then return false end
    local currentVelocity = hrp.Velocity
    local currentHorizSpeed = Vector3.new(currentVelocity.X, 0, currentVelocity.Z).Magnitude
    local lastHorizSpeed = Vector3.new(lastVelocity.X, 0, lastVelocity.Z).Magnitude
    local speedDrop = lastHorizSpeed - currentHorizSpeed
    local wasMovingFast = lastHorizSpeed > 15
    local isInAir = hum.FloorMaterial == Enum.Material.Air
    if speedDrop > 10 and wasMovingFast and isInAir then
        local raycastParams = RaycastParams.new()
        raycastParams.FilterType = Enum.RaycastFilterType.Exclude
        raycastParams.FilterDescendantsInstances = {char}
        local feetPos = getFeetPosition(hrp)
        local forwardDir = hrp.CFrame.LookVector
        local forwardHit = workspace:Raycast(feetPos, forwardDir * 2.5, raycastParams)
        local downHit = workspace:Raycast(feetPos, Vector3.new(0, -1.5, 0), raycastParams)
        if forwardHit then
            if downHit and downHit.Instance == forwardHit.Instance then return false end
            return true
        end
        if speedDrop > 25 and lastHorizSpeed > 20 then return true end
    end
    return false
end

local function launchPlayer(hrp)
    if hasTriggered then return end
    hasTriggered = true
    local moveDir = Vector3.new(lastVelocity.X, 0, lastVelocity.Z)
    if moveDir.Magnitude < 0.1 then moveDir = hrp.CFrame.LookVector else moveDir = moveDir.Unit end
    hrp.Velocity = Vector3.new(moveDir.X * 35, LAUNCH_POWER, moveDir.Z * 35)
end

RunService.Heartbeat:Connect(function()
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    local hum = char and char:FindFirstChild("Humanoid")
    if hrp and hum then
        lastVelocity = hrp.Velocity
        if hum.FloorMaterial ~= Enum.Material.Air and hasTriggered then hasTriggered = false end
    end
end)

task.spawn(function()
    while true do
        task.wait()
        if edgeTrimpEnabled then
            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChild("Humanoid")
            if char and hrp and hum then
                if checkEdgeCollision(char, hrp, hum) and not hasTriggered then launchPlayer(hrp) end
            end
        end
    end
end)

local STRAFE_FORCE = 182

local function InjectCustomAirMove(movementTable)
    if not movementTable or type(movementTable.AirMove) ~= "function" then
        return false
    end
    movementTable.AirMove = function(p129)
        if p129.f.a ~= 0 then
            p129:ApplyFriction(1)
        end

        local v130 = game.Players.LocalPlayer.PlayerScripts.CameraCFrame.Value
        local v131 = p129.f.b
        local v132 = p129.f.a
        local v133 = v130:VectorToWorldSpace((Vector3.new(v131, 0, v132)))

        local v134 = v133.X
        local v135 = v133.Z
        local v136 = Vector3.new(v134, 0, v135)
        local v137 = v136.Magnitude * p129.j
        if v136 ~= Vector3.new(0, 0, 0) then
            v136 = v136.Unit
        end

        local v138 = p129.d:Dot(v136) < 0 and 15 or 20
        local v139

        if p129.f.a == 0 and p129.f.b ~= 0 then
            v139 = v137 > STRAFE_FORCE and STRAFE_FORCE or v137
            v138 = STRAFE_FORCE
        else
            v139 = v137
        end

        p129:Accelerate(v136, v139, v138)
        p129:AirController(v136, v137)
        p129:WallImpact()

        local _ = p129.d.magnitude
        local v140 = p129.Character.Humanoid:GetAttribute("RealSpeed")
        local v141 = 1
        for _, v142 in pairs(p129.Character.StatChanges.Speed:GetChildren()) do
            local v143 = v141 + v142.Value - 1
            v141 = math.clamp(v143, 0, 2.5)
        end
        if p129.Character:GetAttribute("Zombie") == true then
            v141 = v141 * 1.3
        elseif p129.Character:GetAttribute("Nextbot") == nil or p129.Character:GetAttribute("Nextbot") ~= "Angry Munci" and (p129.Character:GetAttribute("Nextbot") ~= "Super Munci" and p129.Character:GetAttribute("Nextbot") ~= "Happy") then
            if p129.Character:GetAttribute("Nextbot") ~= nil and game.Players.LocalPlayer.Character:GetAttribute("PlayerNextbot") == nil then
                v141 = v141 * 2.1
            end
        else
            v141 = v141 * 8
        end
        local v144 = v140 * v141
        local v145 = p129.d.magnitude / (v144 * 2) * 2
        local v146 = p129.z or 2
        p129.c = math.clamp(v145, 1, v146)
    end
    return true
end

local function TryStrafeMemoryScan()
    if not getgc then return false end
    for _, obj in pairs(getgc(true)) do
        if type(obj) == "table" and rawget(obj, "AirMove") and type(obj.AirMove) == "function" then
            if InjectCustomAirMove(obj) then
                return true
            end
        end
    end
    return false
end

local function TryStrafeReplicatedStorageScan()
    for _, obj in pairs(ReplicatedStorage:GetDescendants()) do
        if obj:IsA("ModuleScript") then
            local ok, result = pcall(require, obj)
            if ok and type(result) == "table" and rawget(result, "AirMove") then
                if InjectCustomAirMove(result) then
                    return true
                end
            end
        end
    end
    return false
end

local function RunStrafeInjection(character)
    local movementScript = character:WaitForChild("Movement", 4)
    if movementScript and movementScript:IsA("ModuleScript") then
        local ok, mod = pcall(require, movementScript)
        if ok and InjectCustomAirMove(mod) then
            Notify("Air Strafe", "Injected via Movement module.")
            return
        end
    end
    task.spawn(function()
        for attempts = 1, 10 do
            task.wait(1)
            if TryStrafeMemoryScan() or TryStrafeReplicatedStorageScan() then
                Notify("Air Strafe", "Injected successfully.")
                break
            end
            if attempts == 10 then
                Notify("Air Strafe", "Failed to find AirMove target.")
            end
        end
    end)
end

LocalPlayer.CharacterAdded:Connect(function(char)
    RunStrafeInjection(char)
end)

-- Auto Jump logic
local SCRIPT_ACTIVE = false
local JUMP_HEIGHT = 3
local HITBOX_DISTANCE = 0
local CURRENT_KEYBIND = "SpaceBar"
local storedVelocity = Vector3.new(0, 0, 0)

local function getJumpForce()
    local gravity = workspace.Gravity
    return math.sqrt(2 * gravity * JUMP_HEIGHT)
end

local function isNearGround(character, rootPart)
    if not character or not rootPart then return false end
    local rayParam = RaycastParams.new()
    rayParam.FilterDescendantsInstances = {character}
    rayParam.FilterType = Enum.RaycastFilterType.Exclude
    local ray = workspace:Raycast(rootPart.Position, Vector3.new(0, -(HITBOX_DISTANCE + 3), 0), rayParam)
    return ray ~= nil
end

local function setAutoJumpState(enabled)
    SCRIPT_ACTIVE = enabled
    local character = LocalPlayer.Character
    local humanoid = character and character:FindFirstChild("Humanoid")
    if humanoid then
        if not SCRIPT_ACTIVE then
            humanoid.JumpHeight = 7.2
            Notify("Auto Jump", "BHOP: OFF")
        else
            humanoid.JumpHeight = 0
            Notify("Auto Jump", "BHOP: ON (Distance: " .. HITBOX_DISTANCE .. ")")
        end
    end
end

UserInputService.InputBegan:Connect(function(input, processed)
    if processed then return end
    local targetKey = nil
    pcall(function() targetKey = Enum.KeyCode[CURRENT_KEYBIND] end)
    if targetKey and input.KeyCode == targetKey then
        setAutoJumpState(not SCRIPT_ACTIVE)
    end
end)

RunService.PostSimulation:Connect(function()
    if not SCRIPT_ACTIVE then return end
    local character = LocalPlayer.Character
    local humanoid = character and character:FindFirstChild("Humanoid")
    local rootPart = character and character:FindFirstChild("HumanoidRootPart")
    if not character or not humanoid or not rootPart then return end
    local moveDir = humanoid.MoveDirection
    local isMoving = moveDir.Magnitude > 0
    local state = humanoid:GetState()
    local isAir = (state == Enum.HumanoidStateType.Freefall or state == Enum.HumanoidStateType.Jumping)
    if isMoving then
        if isAir then
            storedVelocity = Vector3.new(rootPart.Velocity.X, 0, rootPart.Velocity.Z)
            if isNearGround(character, rootPart) and rootPart.Velocity.Y <= 0 then
                rootPart.Velocity = Vector3.new(storedVelocity.X, getJumpForce(), storedVelocity.Z)
                humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
            end
        else
            rootPart.Velocity = Vector3.new(storedVelocity.X, getJumpForce(), storedVelocity.Z)
            humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
        end
    end
end)

RunService.Stepped:Connect(function()
    if SCRIPT_ACTIVE then
        local character = LocalPlayer.Character
        local humanoid = character and character:FindFirstChild("Humanoid")
        if humanoid then humanoid.JumpHeight = 0 end

local surfEnabled = false
local SURF_KEYBIND = "U"
local surfActiveConnection = nil

local function setupSurf(character)
    if surfActiveConnection then
        surfActiveConnection:Disconnect()
        surfActiveConnection = nil
    end

    local humanoid = character:WaitForChild("Humanoid", 5)
    local root = character:WaitForChild("HumanoidRootPart", 5)
    if not humanoid or not root then return end

    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = {character}

    local wasSurfing = false
    local DEFAULT_START_SPEED = 25

    surfActiveConnection = RunService.Heartbeat:Connect(function(dt)
        if not character.Parent or not root.Parent or not humanoid.Parent then
            if surfActiveConnection then
                surfActiveConnection:Disconnect()
                surfActiveConnection = nil
            end
            return
        end

        local rayResult = Workspace:Raycast(root.Position, Vector3.new(0, -12, 0), params)
        local currentlyOnRamp = false

        if surfEnabled and rayResult then
            local normal = rayResult.Normal
            local angle = math.deg(math.acos(math.clamp(normal.Y, -1, 1)))

            if angle >= 35 and angle <= 85 then
                currentlyOnRamp = true
                humanoid:SetStateEnabled(Enum.HumanoidStateType.Jumping, false)
                humanoid.PlatformStand = true

                local currentVelocity = root.AssemblyLinearVelocity

                if not wasSurfing then
                    if currentVelocity.Magnitude > DEFAULT_START_SPEED then
                        currentVelocity = currentVelocity.Unit * DEFAULT_START_SPEED
                    end
                end

                local planeVelocity = currentVelocity - (currentVelocity:Dot(normal) * normal)

                local workspaceGravity = (Workspace.Gravity > 0 and Workspace.Gravity or 196.2) * 0.5
                local gravityVector = Vector3.new(0, -workspaceGravity, 0)
                local gravityOnPlane = gravityVector - (gravityVector:Dot(normal) * normal)

                local moveDir = humanoid.MoveDirection
                local strafeForce = Vector3.zero

                if moveDir.Magnitude > 0 then
                    local projectedMove = moveDir - (moveDir:Dot(normal) * normal)
                    if projectedMove.Magnitude > 0 then
                        strafeForce = projectedMove.Unit * 60
                    end
                end

                local finalPlaneVelocity = planeVelocity + ((gravityOnPlane + strafeForce) * dt)

                if finalPlaneVelocity.Magnitude > 350 then
                    finalPlaneVelocity = finalPlaneVelocity.Unit * 350
                end

                root.AssemblyLinearVelocity = finalPlaneVelocity
            else
                humanoid:SetStateEnabled(Enum.HumanoidStateType.Jumping, true)
                humanoid.PlatformStand = false
            end
        else
            humanoid:SetStateEnabled(Enum.HumanoidStateType.Jumping, true)
            humanoid.PlatformStand = false
        end

        wasSurfing = currentlyOnRamp
    end)
end

local MV_BASE_Y      = 0.915
local MV_SLOT_SPACE  = 50
local MV_SLIDE_Y     = 0.015

local mvIndicators = {}

local function mvRecalcLayout()
    local active = {}
    for _, entry in pairs(mvIndicators) do
        if entry.active then table.insert(active, entry) end
    end
    table.sort(active, function(a, b) return a.order < b.order end)
    local n = #active
    if n == 0 then return end
    local totalWidth = (n - 1) * MV_SLOT_SPACE
    for i, entry in ipairs(active) do
        local xOff = -totalWidth / 2 + (i - 1) * MV_SLOT_SPACE
        entry.baseXOffset = xOff
        if entry.label.Visible then
            TweenService:Create(entry.label,
                TweenInfo.new(0.15, Enum.EasingStyle.Quart, Enum.EasingDirection.Out),
                { Position = UDim2.new(0.5, xOff, MV_BASE_Y, 0) }):Play()
        end
    end
end

local function mvSetIndicator(key, isActive)
    local entry = mvIndicators[key]
    if not entry then return end
    entry.active = isActive
    mvRecalcLayout()
    local xOff = entry.baseXOffset or 0
    local label = entry.label
    if isActive then
        label.Visible = true
        label.TextTransparency = 1
        label.Position = UDim2.new(0.5, xOff, MV_BASE_Y + MV_SLIDE_Y, 0)
        TweenService:Create(label,
            TweenInfo.new(0.2, Enum.EasingStyle.Quart, Enum.EasingDirection.Out),
            { TextTransparency = 0, Position = UDim2.new(0.5, xOff, MV_BASE_Y, 0) }):Play()
    else
        local tw = TweenService:Create(label,
            TweenInfo.new(0.2, Enum.EasingStyle.Quart, Enum.EasingDirection.In),
            { TextTransparency = 1, Position = UDim2.new(0.5, xOff, MV_BASE_Y + MV_SLIDE_Y, 0) })
        tw:Play()
        tw.Completed:Connect(function()
            if label.TextTransparency >= 0.99 then label.Visible = false end
        end)
    end
end

-- ── Pixel Surf helpers ────────────────────────────────────
local function mvTouchingWall()
    if not mvState.root then return false end
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = { mvState.char }
    params.FilterType = Enum.RaycastFilterType.Exclude
    for _, dir in ipairs({
        mvState.root.CFrame.LookVector,
        -mvState.root.CFrame.LookVector,
        mvState.root.CFrame.RightVector,
        -mvState.root.CFrame.RightVector,
    }) do
        if workspace:Raycast(mvState.root.Position, dir * MV_CONFIG.WALL_DETECT_DIST, params) then
            return true
        end
    end
    return false
end

local function mvGetCardinalDir()
    local look = Vector3.new(mvCamera.CFrame.LookVector.X, 0, mvCamera.CFrame.LookVector.Z)
    if look.Magnitude == 0 then return Vector3.new(0, 0, 1) end
    look = look.Unit
    if math.abs(look.X) > math.abs(look.Z) then
        return Vector3.new(look.X > 0 and 1 or -1, 0, 0)
    else
        return Vector3.new(0, 0, look.Z > 0 and 1 or -1)
    end
end

-- ── Wall Climb helpers ────────────────────────────────────
local function mvCastWallhopRays(root)
    if not root then return nil end
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = { mvState.char }
    params.FilterType = Enum.RaycastFilterType.Exclude
    local best, bestDist = nil, math.huge
    for i = 0, MV_CONFIG.WALLHOP_NUM_RAYS - 1 do
        local angle = (i / MV_CONFIG.WALLHOP_NUM_RAYS) * math.pi * 2
        local dir   = Vector3.new(math.cos(angle), 0, math.sin(angle))
        local result = workspace:Raycast(root.Position, dir * MV_CONFIG.WALLHOP_RAY_DISTANCE, params)
        if result and result.Distance < bestDist then
            best     = result
            bestDist = result.Distance
        end
    end
    return best
end

local function mvPerformWallhop(root, wallResult)
    local now = tick()
    if now - mvState.lastWallhopTime < MV_CONFIG.WALLHOP_COOLDOWN then return end
    mvState.lastWallhopTime = now
    local vel     = root.AssemblyLinearVelocity
    local hVelVec = Vector3.new(vel.X, 0, vel.Z)
    local hSpeed  = math.max(hVelVec.Magnitude, 22)
    local wn      = Vector3.new(wallResult.Normal.X, 0, wallResult.Normal.Z)
    if wn.Magnitude > 0.01 then wn = wn.Unit end
    local reflected = hVelVec - 2 * hVelVec:Dot(wn) * wn
    local newDir    = reflected.Magnitude > 0.1 and reflected.Unit or wn
    mvState.isWallhopFlicking = true
    task.delay(MV_CONFIG.WALLHOP_JUMP_DELAY, function()
        if mvState.hum and mvState.hum.Health > 0 then
            mvState.hum:ChangeState(Enum.HumanoidStateType.Jumping)
            root.AssemblyLinearVelocity = Vector3.new(
                newDir.X * hSpeed, MV_CONFIG.WC_BOOST_HEIGHT, newDir.Z * hSpeed)
        end
        task.delay(MV_CONFIG.WALLHOP_FLICK_DURATION, function()
            mvState.isWallhopFlicking = false
        end)
    end)
end

    end
    if kc == mvKeybinds.wallClimbV2 and mvFeatures.wallClimbV2 then

            local wallResult = mvCastWallhopRays(mvState.root)
            if wallResult then
                local vel     = mvState.root.AssemblyLinearVelocity
                local hVelVec = Vector3.new(vel.X, 0, vel.Z)
                local hSpeed  = math.max(hVelVec.Magnitude, 22)
                local wn      = Vector3.new(wallResult.Normal.X, 0, wallResult.Normal.Z)
                if wn.Magnitude > 0.01 then wn = wn.Unit end
                local reflected = hVelVec - 2 * hVelVec:Dot(wn) * wn
                local newDir    = reflected.Magnitude > 0.1 and reflected.Unit or wn
                mvState.hum:ChangeState(Enum.HumanoidStateType.Jumping)
                mvState.root.AssemblyLinearVelocity = Vector3.new(
                    newDir.X * hSpeed, 52, newDir.Z * hSpeed)
            end
        end
    end
end)

-- ── Main loop (RenderStepped) ─────────────────────────────
RunService.RenderStepped:Connect(function(dt)
    local curPx  = mvFeatures.pixelSurf == true and mvState.keys.Surf == true
    local curAir = mvFeatures.airstuck  == true and mvState.keys.Airstuck == true
    local curWc  = mvFeatures.autoWallhop == true and mvState.wallhopHolding == true

    if curPx  ~= mvLastPx  then mvLastPx  = curPx  mvSetIndicator("px",  curPx)  end
    if curAir ~= mvLastAir then mvLastAir = curAir mvSetIndicator("air", curAir) end
    if curWc  ~= mvLastWc  then mvLastWc  = curWc  mvSetIndicator("wc",  curWc)  end

    if not mvState.hum or not mvState.root then return end

    if mvFeatures.unstablePixelSurf and mvState.keys.Surf and mvTouchingWall() then
        if not mvState.isSurfing then
            mvState.isSurfing         = true
            mvState.surfY             = mvState.root.Position.Y
            mvState.unstableSurfTimer = 0
            mvState.isUnstableFalling = false
        end
        mvState.unstableSurfTimer = mvState.unstableSurfTimer + dt
        local dir = mvGetCardinalDir()
        if mvState.isUnstableFalling then
            if mvState.unstableSurfTimer > 0.25 then
                mvState.isUnstableFalling = false
                mvState.unstableSurfTimer = 0
                mvState.surfY = mvState.root.Position.Y
            else
                mvState.root.AssemblyLinearVelocity = Vector3.new(
                    dir.X * MV_CONFIG.PIXELSURF_SPEED, -25, dir.Z * MV_CONFIG.PIXELSURF_SPEED)
            end
        else
            if mvState.unstableSurfTimer > 1.2 then
                mvState.isUnstableFalling = true
                mvState.unstableSurfTimer = 0
            else
                mvState.root.AssemblyLinearVelocity = Vector3.new(
                    dir.X * MV_CONFIG.PIXELSURF_SPEED, 0, dir.Z * MV_CONFIG.PIXELSURF_SPEED)
                mvState.root.CFrame = CFrame.new(
                    mvState.root.Position.X, mvState.surfY, mvState.root.Position.Z)
                    * (mvState.root.CFrame - mvState.root.CFrame.Position)
            end
        end

    elseif mvFeatures.pixelSurf and mvState.keys.Surf and mvTouchingWall() then
        if not mvState.isSurfing then
            mvState.isSurfing = true
            mvState.surfY     = mvState.root.Position.Y
        end
        local dir = mvGetCardinalDir()
        mvState.root.AssemblyLinearVelocity = Vector3.new(
            dir.X * MV_CONFIG.PIXELSURF_SPEED, 0, dir.Z * MV_CONFIG.PIXELSURF_SPEED)
        mvState.root.CFrame = CFrame.new(
            mvState.root.Position.X, mvState.surfY, mvState.root.Position.Z)
            * (mvState.root.CFrame - mvState.root.CFrame.Position)
    else
        mvState.isSurfing         = false
        mvState.isUnstableFalling = false
    end
end)

-- ── Heartbeat ─────────────────────────────────────────────
RunService.Heartbeat:Connect(function()
    -- Airstuck
    local shouldAirstuck = mvFeatures.airstuck and mvState.keys.Airstuck
    if shouldAirstuck then
        if not mvState.isAirstuckActive then
            mvState.isAirstuckActive = true
            if mvState.root then
                local bp        = Instance.new("BodyPosition")
                bp.Name         = "AirstuckLock"
                bp.MaxForce     = Vector3.new(1e9, 1e9, 1e9)
                bp.P            = 1e5
                bp.D            = 1e4
                bp.Position     = mvState.root.Position
                bp.Parent       = mvState.root
                mvState.airstuckBodyPos = bp
            end
        end
        if mvState.root then
            mvState.root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        end
    else
        if mvState.isAirstuckActive then
            mvState.isAirstuckActive = false
            if mvState.airstuckBodyPos and mvState.airstuckBodyPos.Parent then
                mvState.airstuckBodyPos:Destroy()
            end
            mvState.airstuckBodyPos = nil
        end
    end

    -- autoWallhop
    if mvFeatures.autoWallhop and mvState.wallhopHolding
        and not mvState.isWallhopFlicking
        and not mvState.menuOpen then
        if mvState.root and mvState.hum and mvState.hum.Health > 0 then
            local wallResult = mvCastWallhopRays(mvState.root)
            if wallResult then mvPerformWallhop(mvState.root, wallResult) end
        end
    end
end)

-- Unlock Strafe Locked Emotes logic
local strafeUnlockEnabled = false
local strafeConnection = nil
local strafeChildConnection = nil

local function setupStrafeUnlock()
    local playerName = LocalPlayer.Name
    local success, speedFolder = pcall(function()
        return workspace:WaitForChild("Game"):WaitForChild("Players"):WaitForChild(playerName):WaitForChild("StatChanges"):WaitForChild("Speed")
    end)
    if not success or not speedFolder then
        Notify("Strafe Unlock", "Speed folder not found! Make sure you're in-game.")
        return false
    end
    local function setEmoteSpeedToOne(child)
        if child.Name == "EmoteSpeed" and child:IsA("NumberValue") then child.Value = 1 end
    end
    if strafeConnection then strafeConnection:Disconnect() end
    if strafeChildConnection then strafeChildConnection:Disconnect() end
    for _, child in ipairs(speedFolder:GetChildren()) do
        if child.Name == "EmoteSpeed" and child:IsA("NumberValue") then child.Value = 1 end
    end

    return UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled and not UserInputService.MouseEnabled

local function stopOverhaulAnimation()
    if overhaulAnimTrack then
        overhaulAnimTrack:Stop()
        overhaulAnimTrack = nil
    end
end

local function playOverhaulAnimationLoop()
    while Character and Character.Parent and isOverhaulActive do
        local humanoid = Character:FindFirstChild("Humanoid")
        if not humanoid then break end
        local animator = humanoid:FindFirstChild("Animator")
        if not animator then break end
        for _, track in pairs(animator:GetPlayingAnimationTracks()) do
            if track.Name == "ForceAnimation" then track:Stop() end
        end
        local animation = Instance.new("Animation")
        animation.AnimationId = "rbxassetid://" .. selectedEmoteID
        animation.Name = "ForceAnimation"
        overhaulAnimTrack = animator:LoadAnimation(animation)
        overhaulAnimTrack.Looped = true
        overhaulAnimTrack:Play()
        while Character and Character.Parent and isOverhaulActive and overhaulAnimTrack and overhaulAnimTrack.IsPlaying do
            task.wait(0.1)
        end
        task.wait(0.1)
    end
    stopOverhaulAnimation()
end

local function createOverhaulEmoteSpeed()
    local myPlayer = workspace.Game.Players:FindFirstChild(LocalPlayer.Name)
    if myPlayer then
        local statChanges = myPlayer:FindFirstChild("StatChanges")
        if not statChanges then statChanges = Instance.new("Folder"); statChanges.Name = "StatChanges"; statChanges.Parent = myPlayer end
        local speed = statChanges:FindFirstChild("Speed")
        if not speed then speed = Instance.new("Folder"); speed.Name = "Speed"; speed.Parent = statChanges end
        local emoteSpeed = speed:FindFirstChild("EmoteSpeed")
        if not emoteSpeed then
            emoteSpeed = Instance.new("NumberValue"); emoteSpeed.Name = "EmoteSpeed"; emoteSpeed.Value = 1; emoteSpeed.Parent = speed
        else
            emoteSpeed.Value = 1
        end
        return true
    end
    return false
end

local function removeOverhaulEmoteSpeed()
    local myPlayer = workspace.Game.Players:FindFirstChild(LocalPlayer.Name)
    if myPlayer then
        local statChanges = myPlayer:FindFirstChild("StatChanges")
        if statChanges then
            local speed = statChanges:FindFirstChild("Speed")
            if speed then
                local emoteSpeed = speed:FindFirstChild("EmoteSpeed")
                if emoteSpeed then emoteSpeed:Destroy() end
            end
        end
    end
end

local function iniciarBloqueioOverhaulAnimacoes()
    if not Character then return end
    local humanoid = Character:FindFirstChild("Humanoid")
    if not humanoid then return end
    local animator = humanoid:FindFirstChild("Animator")
    if not animator then return end
    local animsParaRemover = {"Crouch", "Slide"}
    local function verificarETreinarTrack(track)
        if track and track.Stop then
            local anim = track.Animation
            if anim then
                for _, palavra in pairs(animsParaRemover) do
                    if string.find(anim.Name or "", palavra) or string.find(anim.AnimationId or "", palavra) then
                        track:Stop(); track:Destroy(); break
                    end
                end
            end
        end
    end
    for _, track in pairs(animator:GetPlayingAnimationTracks()) do verificarETreinarTrack(track) end
    if overhaulAnimBlockConnection then overhaulAnimBlockConnection:Disconnect() end
    overhaulAnimBlockConnection = animator.AnimationPlayed:Connect(verificarETreinarTrack)
end

local function pararBloqueioOverhaulAnimacoes()
    if overhaulAnimBlockConnection then overhaulAnimBlockConnection:Disconnect(); overhaulAnimBlockConnection = nil end
end

        end)
    end
    overhaulLookYConnection = RunService.Heartbeat:Connect(function()

    end)
    forceAttribute()
end

local function aplicarOverhaulConfiguracoesAtivacao()
    local player = LocalPlayer

    if not targetPlayer then
        for _, otherPlayer in pairs(workspace.Game.Players:GetChildren()) do
            if otherPlayer.Name ~= player.Name then targetPlayer = otherPlayer; break end
        end
    end
    if targetPlayer then
        player.CameraMaxZoomDistance = 10
        player.CameraMinZoomDistance = 10
        player.CameraMode = Enum.CameraMode.Classic
        if player:FindFirstChild("PlayerScripts") then
            local mlo = player.PlayerScripts:FindFirstChild("MouseLockOffset")
            if mlo then mlo.Value = Vector3.new(0,0,0) end
        end
        targetPlayer:SetAttribute("FirstPerson", false)
    end
    ativarOverhaulShiftLock()
end

        end

    pararForcadorOverhaulLookY()

end

-- Accessory ID Spawn logic
local function addAccessoryToCharacter(accessoryId, parentPart)
    local success, assets = pcall(function() return game:GetObjects("rbxassetid://" .. tostring(accessoryId)) end)
    if not success or #assets == 0 then Notify("Error", "Invalid accessory ID!"); return end
    local accessory = assets[1]
    local character = LocalPlayer.Character
    accessory.Parent = game.Workspace
    local handle = accessory:FindFirstChild("Handle")
    if handle then
        handle.CanCollide = false
        local weld = Instance.new("Weld")
        weld.Part0 = parentPart; weld.Part1 = handle
        weld.C0 = parentPart.CFrame:Inverse() * handle.CFrame
        weld.Parent = parentPart
    end
    accessory.Parent = character
    Notify("Accessory", "Spawned successfully!")
end
